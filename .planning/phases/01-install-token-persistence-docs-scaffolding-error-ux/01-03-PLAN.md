```typescript
const isTauri = typeof window !== 'undefined' && !!(window.__TAURI_INTERNALS__ || window.__TAURI__);

export function getApiBase(): string {
  // 1. Explicit override always wins (Docker users / development)
  if (import.meta.env.VITE_OMNIVOICE_API) return import.meta.env.VITE_OMNIVOICE_API;
  // 2. Inside Tauri webview: backend is on the host machine's localhost.

}
```

      ```
      import os
      import subprocess
      import sys
      import logging
      from typing import Optional

      logger = logging.getLogger("omnivoice.gatekeeper")

      def _resolve_app_bundle_path() -> Optional[str]:
          # When running inside an installed .app, sys.executable points into
          # the bundle's Resources or MacOS directory. Walk up to find .app root.
          if sys.platform != "darwin":
              return None
          path = os.path.realpath(sys.executable)
          while path != "/":
              if path.endswith(".app"):
                  return path
              path = os.path.dirname(path)
          return None

      def is_app_quarantined() -> bool:
          if sys.platform != "darwin":
              return False
          bundle = _resolve_app_bundle_path()
          if not bundle:
              return False  # dev run or non-bundled; treat as not quarantined
          try:
              result = subprocess.run(
                  ["xattr", "-l", bundle],
                  capture_output=True,
                  text=True,
                  timeout=5,
              )
          except (FileNotFoundError, subprocess.TimeoutExpired) as e:
              logger.warning("xattr probe failed: %s", e)
              return False
          return "com.apple.quarantine" in result.stdout
      ```
